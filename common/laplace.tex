\section{Operator Definition}
%=======================================
%--------------------------------------
\begin{definition}
\footnote{
  \citerpgc{bracewell1978}{219}{007007013X}{Chapter 11 The Laplace transform},
  \citerpcu{vanderpol1959}{13}{5. Strip of convergence of the Laplace integral}{https://archive.org/details/in.ernet.dli.2015.141269/page/n23/},
  \citerpcu{levy1958}{2}{``two-sided transformation"}{https://archive.org/details/c_cowles_ieee_Levy/page/n5/mode/2up},
  \citerpcu{betten2008L}{295}{(B.1)}{https://link.springer.com/content/pdf/bbm:978-3-540-85051-9/1.pdf}
  }
\label{def:opLT}
%--------------------------------------
Let $\spLLRBu$ be the space of all \structe{Lebesgue square-integrable functions}.
\defboxt{
  The \opd{Laplace Transform} operator $\opLT$ is here defined as
  \\\indentx$\ds
    \brs{\hxs{\opLT} \ff}(s)
    \eqd \Lscale
    \int_{x\in\R} \ff(x) e^{-sx} \dx
    \qquad\scy\forall \ff\in\spLLRBu
  $
  %\\This definition of the Fourier Transform is also called the \opd{unitary Laplace Transform}.
  }
\end{definition}

Such integrals may \prope{converge} for certain values of $s$ and \prope{diverge} for others.
%--------------------------------------
\begin{definition}
\label{def:roc}
%--------------------------------------
Let $\opLT\brs{\fg(x)}$ be the \ope{Laplace Transform} \xref{def:opLT} of a function $\fg(x)$.
\defboxt{
  The set $\opROC\opLT\brs{\fg(x)}$ of all $s$ for which $\opLT\brs{\fg(x)}$ \prope{converges}
  \\is the \structd{Region of Convergence} of $\opLT\brs{\fg(x)}$.
  }
\end{definition}

In this text, the region of convergence may in places be specified using
the \structe{open interval} $\intoo{A}{B}$ and \structe{closed interval}
$\intcc{A}{B}$\ifsxref{convex}{def:intcc}.

%--------------------------------------
\begin{remark}
%--------------------------------------
A scaling factor $\frac{1}{\sqrt{2\pi}}$ in front of $\int_\R$ in \pref{def:opLT}
is not typically found in references offering
definitions of the Laplace Transform, and is not included here either.
That is not to say, however, that it's not a good idea.
Including it would make the operator $\opLT$ more directly compatible
with the \ope{Unitary Fourier Transform} operator $\opFT$\ifsxref{harFour}{def:opFT}.
Note also that a $\frac{1}{2\pi}$ scaling factor is included in
[\citerpg{bachman2002}{268}{9780387988993}] in their definition of \ope{convolution}
\xxref{def:conv}{sec:opLT_conv}.
\end{remark}

%=======================================
%--------------------------------------
\begin{theorem}
\footnote{
  \citerpgc{bracewell1978}{220}{007007013X}{Chapter 11 The Laplace transform}
  }
\label{thm:opLTi}
%--------------------------------------
\thmbox{\begin{array}{rc>{\ds}lM}
  \fg(x) &=& \opLTi[\fG(s)] \eqd \frac{1}{2\pi i}\int_{c-i\infty}^{c+i\infty} \fG(s) e^{sx} ds
           & for some $c\in\Rp$
\end{array}}
\end{theorem}

\section{Transversal properties}
%=======================================
%--------------------------------------
\begin{theorem}
\footnote{
  \citerpgc{bracewell1978}{224}{007007013X}{Table 11.1: ``Shift" and ``Similarity" entries},
  \citerpcu{levy1958}{15}{Equation 0.8}{https://archive.org/details/c_cowles_ieee_Levy/page/n15/}
  }
\label{thm:opLT_transversal}
\label{thm:opLT_shift}
\label{thm:opLT_dilation}
%--------------------------------------
Let $\opLT\brs{\fg(x)}$ be the \ope{Laplace Transform} \xref{def:opLT} of a function $\fg(x)$.
Let the \structe{Region of Convergence} of $\opLT[\fg(x)](s)$
be $A\leq\Real(s)\leq B$ with $\opair{A}{B}\in\R^2$.
\thmbox{\begin{array}{>{\ds}lc>{\ds}r Dl@{\hspace{2pt}}l CD}
    \mc{3}{H}{Mapping}                                                                         & \mc{3}{|H}{Region of Convergence}                   & \mc{1}{|H}{Domain}    & Property
  \\\hline%                  & &                                                               &     &                             &                 &                       &
    \opLT\brs{\fg(x-\alpha)} &=& e^{-\alpha s}         \opLT\brs{\fg(x)}\brp{s}                & for & \Real(s)                    & \in\intcc{A}{B} & \forall x,\alpha\in\C & (\ope{translation})
  \\\opLT\brs{\fg(\alpha x)} &=& \frac{1}{\abs{\alpha}}\opLT\brs{\fg(x)}\brp{\frac{s}{\alpha}} & for & \Real\brp{\frac{s}{\alpha}} & \in\intcc{A}{B} & \forall x,\alpha\in\C & (\ope{dilation})
\end{array}}
\end{theorem}
\begin{proof}
\begin{align*}
  \opLT\brs{\fg(x-\alpha)}
    &\eqd \Lscale \int_{x=-\infty}^{x=\infty} \fg(x-\alpha) e^{-sx} \dx                 &&                              && \text{by definition of $\opLT$} && \text{\xref{def:opLT}}
  \\&=    \Lscale \int_{u+\alpha=-\infty}^{u+\alpha=\infty} \fg(u) e^{-s(\alpha+u)} \du &&                              && \text{where $u\eqd x-\alpha$}   && \text{$\implies$ $x=\alpha+u$}
  \\&=    e^{-\alpha s} \;\Lscale \int_{u=-\infty}^{u=\infty} \fg(u) e^{-su} \du        && \forall A \leq\Real(s)\leq B && \text{by property of exponents} && \text{$b^{x+\alpha}=b^xb^\alpha$}
  \\&\eqd e^{-\alpha s} \;\Lscale \int_{x=-\infty}^{x=\infty} \fg(x) e^{-sx} \dx        && \forall A \leq\Real(s)\leq B && \text{by change of variable}    && \text{$u\to x$}
  \\&\eqd e^{-\alpha s} \;\brs{\opLT \fg(x)}                                            && \forall A \leq\Real(s)\leq B && \text{by definition of $\opLT$} && \text{\xref{def:opLT}}
  \\
  \\
  \opLT\brs{\fg(\alpha x)}
    &\eqd \Lscale \int_{x=-\infty}^{x=\infty} \fg(\alpha x) e^{-sx} \dx
    &&
    && \text{by definition of $\opLT$}
    && \text{\xref{def:opLT}}
  \\&= \Lscale \int_{u/\alpha=-\infty}^{u/\alpha=\infty} \fg(u) e^{-s(u/\alpha)} \frac{1}{\alpha}\du
    &&
    && \text{where $u\eqd \alpha x$}
    && \text{$\implies$ $x=\frac{u}{\alpha}$}
  \\&= \Lscale \frac{1}{\alpha}\int_{u/\alpha=-\infty}^{u/\alpha=\infty} \fg(u) e^{-(s/\alpha)u} \du
    &&
    &&
    &&
  \\&=\mathrlap{
      \brbl{\begin{array}{>{\ds}lMl}
               \Lscale \frac{1}{\alpha}\int_{u=-\infty}^{u=\infty} \fg(u) e^{-(s/\alpha)u} \du & if $\alpha\ge0$ & \forall A \leq\Real\brp{\frac{s}{\alpha}}\leq B
               \\[2ex]
               \Lscale \frac{1}{\alpha}\int_{u=\infty}^{u=-\infty} \fg(u) e^{-(s/\alpha)u} \du & otherwise       & \forall A \leq\Real\brp{\frac{s}{\alpha}}\leq B
             \end{array}}}
  \\&=\mathrlap{
      \brbl{\begin{array}{>{\ds}rMl}
               \Lscale \frac{1}{\alpha}\int_{u=-\infty}^{u=\infty} \fg(u) e^{-(s/\alpha)u} \du & if $\alpha\ge0$ & \forall A \leq\Real\brp{\frac{s}{\alpha}}\leq B
               \\[2ex]
              -\Lscale \frac{1}{\alpha}\int_{u=-\infty}^{u=\infty} \fg(u) e^{-(s/\alpha)u} \du & otherwise       & \forall A \leq\Real\brp{\frac{s}{\alpha}}\leq B
            \end{array}}}
  \\&=    \frac{1}{\abs{\alpha}} \Lscale \int_{x\in\R} \fg(x) e^{-(s/\alpha)x} \dx   && \forall A \leq\Real\brp{\frac{s}{\alpha}}\leq B && \text{by change of variable}    && u\to x
  \\&\eqd \frac{1}{\abs{\alpha}}\brs{\opLT \fg(x)}\brp{\frac{s}{\alpha}}             && \forall A \leq\Real\brp{\frac{s}{\alpha}}\leq B && \text{by definition of $\opLT$} && \text{\xref{def:opLT}}
\end{align*}
\end{proof}

%--------------------------------------
\begin{corollary}
\footnote{
  \citerpgc{bracewell1978}{224}{007007013X}{Table 11.1 Theorems for the Laplace Transform}
  }
\label{cor:opLT_reversal}
%--------------------------------------
Let $\opLT$, $\fG(s)$, $A$, and $B$ be defined as in \prefpp{thm:opLT_transversal}.
\corbox{\begin{array}{>{\ds}lc>{\ds}l   D       l              @{\hspace{2pt}}l     C                       D}
    \mc{3}{H}{Mapping}                  & \mc{3}{|H}{Region of Convergence}       & \mc{1}{|H}{Domain}    & \mc{1}{|H}{Property}
  \\\hline%                             &     &              &                    &                       &
    \opLT\brs{\fg(-x)}  &=& \fG\brp{-s} & for & \Real\brp{s} & \in\intcc{-B}{-A}  & \forall x,\alpha\in\C & (\ope{reversal})
\end{array}}
\end{corollary}
\begin{proof}
\begin{align*}
  \opLT\brs{\fg(-x)}
    &= \opLT\brs{\fg([-1]x)}
    && \Real\brp{s}\in\intcc{A}{B}
    && \text{by definition of unary operator $-$}
  \\&= \opLT\brs{\frac{1}{\abs{-1}}\fg\brp{\frac{x}{-1}}}
    && \Real\brp{\frac{s}{-1}} \in\intcc{A}{B}
    && \text{by \prope{dilation} property \xref{thm:opLT_dilation}}
  \\&= \fG\brp{-s}
    && \Real\brp{s} \in\intcc{-B}{-A}
    &&
\end{align*}
\end{proof}

\section{Linear properties}
%=======================================
%--------------------------------------
\begin{theorem}
\footnote{
  \citerpgc{bracewell1978}{224}{007007013X}{Table 11.1 Theorems for the Laplace Transform},
  \citerpcu{betten2008H}{296}{(B.6)}{https://link.springer.com/content/pdf/bbm:978-3-540-85051-9/1.pdf},
  \citerpcu{levy1958}{13}{Equation 0.2}{https://archive.org/details/c_cowles_ieee_Levy/page/n15/},
  \citerpcu{vanderpol1959}{22}{Introduction}{https://archive.org/details/in.ernet.dli.2015.141269/page/n31/},
  \citerpgc{shafii2015}{7}{9780738672458}{Theorem 1.4}
  }
\label{thm:opLT_linear}
%--------------------------------------
Let $\opLT$ be the \ope{Laplace Transform} operator \xref{def:opLT}.
Let $\fG(s)\eqd \brs{\opLT\fg(x)}$
and $\fF(s)\eqd \brs{\opLT\ff(x)}$.
Let the \structe{Region of Convergence} of $\fG(s)$ be $A\leq\Real(s)\leq B$
and the \structe{Region of Convergence} of $\fF(s)$ be $C\leq\Real(s)\leq D$.
\thmbox{\begin{array}{>{\ds}l        c>{\ds}l             D     l              @{\hspace{2pt}}l                                  C                       D}
    \mc{3}{H}{Mapping}                                  & \mc{3}{|H}{Region of Convergence}                                    & \mc{1}{|H}{Domain}    & \mc{1}{|H}{Property}
  \\\hline%                         & &                 &                                   &                                  &                       &
    \opLT\brs{\ff(x)+\fg(x)}        &=& \fF(s) + \fG(s) & for & \Real(s)                    & \in\intcc{A}{B}\seti\intcc{C}{D} & \forall x       \in\C & (\prope{additive})
  \\\opLT\brs{\alpha\fg(x)}         &=& \alpha\fG(s)    & for & \Real(s)                    & \in\intcc{A}{B}                  & \forall x,\alpha\in\C & (\prope{homogeneous})
\end{array}}
\end{theorem}

%--------------------------------------
\begin{corollary}[Linear Properties]
\label{cor:opLT_linear}
%--------------------------------------
Let $\opLT$ be the \ope{Laplace Transform} operator \xref{def:opLT}.
Let $A$ and $B$ be real numbers such that $\intcc{A}{B}$
is the \structe{Region of Convergence} of $\opLT\brs{\fg(x)}$.
Let $C$ and $D$ be real numbers such that $\intcc{C}{D}$
is the \structe{Region of Convergence} of $\opLT\brs{\ff(x)}$.
Let $A_n$ and $B_n$ be real numbers such that $\intcc{A_n}{B_n}$
is the \structe{Region of Convergence} of $\opLT\brs{\fg_n(x)}$.
\corbox{\begin{array}{>{\ds}l                      c  >{\ds}l                                       D     >{\ds}l                                               C}
    \mc{3}{H}{Mapping}                                                                            & \mc{2}{|H}{Region of Convergence}                         & \mc{1}{|H}{Domain}
  \\\hline
    \opLT\brs{ 0                               }  &=&  0                                          & for & \Real(s)\in\intcc{A}{B}                             & \forall x\in\C
  \\\opLT\brs{ -\fg(x)                         }  &=& -\opLT\brs{\fg(x)}                          & for & \Real(s)\in\intcc{A}{B}                             & \forall x\in\C
  \\\opLT\brs{ \ff(x)-\fg(x)                   }  &=&  \opLT\brs{\fg(x)} - \opLT\brs{\ff(x)}      & for & \Real(s)\in\intcc{A}{B}\seti\intcc{C}{D}            & \forall x\in\C
  \\\opLT\brs{ \sum_{n=1}^\xN \alpha_n\fg_n(x) }  &=&  \sum_{n=1}^\xN \alpha_n\opLT\brs{\fg_n(x)} & for & \Real(s)\in\setopi_{n=1}^\xN\intcc{A_n}{B_n}        & \forall x,\alpha_n\in\C
\end{array}}
\end{corollary}
\begin{proof}
\begin{enumerate}
  \item By \prefpp{thm:opLT_linear}, the operator \ope{Laplace Transform} operator $\opLT$
        is \prope{additive} and \prope{homogeneous}. \label{item:opLT_linear}
  \item By \pref{item:opLT_linear} and \prefpp{def:linop}, $\opLT$ is \prope{linear}. \label{item:opLT_linear_linear}
  \item By \pref{item:opLT_linear_linear} and \prefpp{thm:L_prop}, the four properties listed follow.
\end{enumerate}
\end{proof}

\section{Modulation properties}
%=======================================
%--------------------------------------
\begin{theorem}
\footnote{
  \citerpgc{bracewell1978}{224}{007007013X}{Table 11.1: ``Modulation" entry},
  \citerpcu{levy1958}{19}{Equation 1.2}{https://archive.org/details/c_cowles_ieee_Levy/page/n19/}
  }
\label{thm:opLT_mod}
%--------------------------------------
Let $\opLT$ be the \ope{Laplace Transform} operator \xref{def:opLT}.
Let $\fG(s)\eqd \brs{\opLT\fg(x)}$.
Let the \structe{Region of Convergence} of $\fG(s)$ be $A\leq\Real(s)\leq B$.
\thmbox{\begin{array}{>{\ds}lc>{\ds}l                   Dl                                                         C                       D}
    \mc{3}{H}{Mapping}                                & \mc{2}{|H}{Region of Convergence}                        & \mc{1}{|H}{Domain}    & \mc{1}{|H}{Property}\\\hline
    \opLT\brs{e^{-\alpha x} \fg(x)} &=& \fG(s+\alpha) & for & A-\Real(\alpha) \leq \Real(s) \leq B-\Real(\alpha) & \forall x,\alpha\in\C & (\ope{modulation})
\end{array}}
\end{theorem}
\begin{proof}
\begin{align*}
  \opLT\brs{e^{-\alpha x} \fg(x)}
    &\eqd \Lscale \int_{x\in\R} e^{-\alpha x} \fg(x) e^{-sx} \dx
          %&& \forall \Real(s)\in\intcc{A-\Real(\alpha)}{B-\Real(\alpha)}
          && \text{by definition of $\opLT$}
          && \text{\xref{def:opLT}}
  \\&=    \Lscale \int_{x\in\R} \fg(x) e^{-(s+\alpha)x} \dx        && A\leq\Real(s+\alpha)\leq B                      && \text{$b^{x+y} = b^x b^y$}
  \\&\eqd \brs{\opLT\fg(x)}(s+\alpha)                              && A-\Real(\alpha)\leq\Real(s)\leq B-\Real(\alpha) && \text{\xref{def:opLT}}
  \\&\eqd \fG(s+\alpha)                                            && A-\Real(\alpha)\leq\Real(s)\leq B-\Real(\alpha) && \text{by definition of $\fG(s)$}
\end{align*}
\end{proof}

%--------------------------------------
\begin{corollary}
\footnote{
  \citerpgc{bracewell1978}{224}{007007013X}{Table 11.1 Theorems for the Laplace Transform}
  }
\label{cor:opLT_mod}
%--------------------------------------
Let $\opLT$ be the \ope{Laplace Transform} operator \xref{def:opLT}.
Let $\fG(s)\eqd \brs{\opLT\fg(x)}$
Let the \structe{Region of Convergence} of $\fG(s)$ be $A\leq\Real(s)\leq B$.
\corbox{\begin{array}{>{\ds}l@{\hspace{2pt}}c@{\hspace{2pt}}>{\ds}r                                          Dl@{\hspace{2pt}}l                                                              C                    } %D
    \mc{3}{H}{Mapping}                                                                                     & \mc{3}{|H}{Region of Convergence}                                             & \mc{1}{|H}{Domain} \\\hline %& \mc{1}{|H}{Property}     \\\hline
    \opLT\brs{\cos \brp{\omega_o x} \fg(x)} &=&  \frac{1}{2}\fG(s-i\omega_o) + \frac{1}{2}\fG(s+i\omega_o) & for & \Real(s) & \in\intcc{A}{B}                                              & \forall x\omega_o\in\C     \\[2ex]  %& (\ope{cosine modulation})\\[2ex]
    \opLT\brs{\sin \brp{\omega_o x} \fg(x)} &=& -\frac{i}{2}\fG(s-i\omega_o) + \frac{i}{2}\fG(s+i\omega_o) & for & \Real(s) & \in\intcc{A}{B}                                              & \forall x\omega_o\in\C     \\[2ex]  %& (\ope{sine modulation})  \\[2ex]
    \opLT\brs{\cosh\brp{\omega_o x} \fg(x)} &=&  \frac{1}{2}\fG(s- \omega_o) + \frac{1}{2}\fG(s+ \omega_o) & for & \Real(s) & \in\intcc{A+\abs{\Real(\omega_o)}}{B-\abs{\Real(\omega_o)}}  & \forall x\omega_o\in\C     \\[2ex]  %& (\ope{cosh modulation})  \\[2ex]
    \opLT\brs{\sinh\brp{\omega_o x} \fg(x)} &=&  \frac{1}{2}\fG(s- \omega_o) - \frac{1}{2}\fG(s+ \omega_o) & for & \Real(s) & \in\intcc{A+\abs{\Real(\omega_o)}}{B-\abs{\Real(\omega_o)}}  & \forall x\omega_o\in\C              %& (\ope{sinh modulation})
\end{array}}
\end{corollary}
\begin{proof}
\begin{enumerate}
  \item Mappings:
    \begin{align*}
      \opLT\brs{\cosh\brp{\omega_o x} \fg(x)}
        &= \opLT\brs{\brp{\frac{e^{\omega_o x} + e^{-\omega_o x}}{2}} \fg(x)}
        && \text{by definition of $\cosh(x)$}
        && \text{\xref{def:cosh}}
      \\&= \frac{1}{2}\opLT\brs{e^{ \omega_o x} \fg(x)}(s)
         + \frac{1}{2}\opLT\brs{e^{-\omega_o x} \fg(x)}(s)
        && \text{by \prope{additive} property}
        && \text{\xref{thm:opLT_linear}}
      \\&=     \frac{1}{2}\opLT\brs{\fg(x)}(s-\omega)
             + \frac{1}{2}\opLT\brs{\fg(x)}(s+\omega)
        && \text{by \prope{modulation} prop.}
        && \text{\xref{thm:opLT_mod}}
      \\&=   \frac{1}{2}\fG(s-\omega_o) + \frac{1}{2}\fG(s+\omega_o)
        && \text{by definition of $\fG(s)$}
        &&
      \\
      \\
      \opLT\brs{\sinh\brp{\omega_o x} \fg(x)}
        &= \opLT\brs{\brp{\frac{e^{\omega_o x} - e^{-\omega_o x}}{2}} \fg(x)}
        && \text{by definition of $\sinh(x)$}
        && \text{\xref{def:sinh}}
      \\&=     \frac{1}{2}\opLT\brs{e^{ \omega_o x} \fg(x)}(s)
             - \frac{1}{2}\opLT\brs{e^{-\omega_o x} \fg(x)}(s)
        && \text{by \prope{additive} property}
        && \text{\xref{thm:opLT_linear}}
      \\&=     \frac{1}{2}\opLT\brs{\fg(x)}(s-\omega)
             - \frac{1}{2}\opLT\brs{\fg(x)}(s+\omega)
        && \text{by \prope{modulation} prop.}
        && \text{\xref{thm:opLT_mod}}
      \\&=   \frac{1}{2}\fG(s-\omega_o) - \frac{1}{2}\fG(s+\omega_o)
        && \text{by definition of $\fG(s)$}
        &&
      \\
      \\
      \opLT\brs{\cos\brp{\omega_o x} \fg(x)}
        &= \opLT\brs{\cosh\brp{i\omega_o x} \fg(x)}
       %&&
        && \text{by \prefp{thm:cosh}}
      \\&=  \frac{1}{2}\fG(s-i\omega_o) + \frac{1}{2}\fG(s+i\omega_o)
        && \text{by $\opLT\brs{\cos(\omega_o x)\fg(x)}$ result}
      \\
      \\
      \opLT\brs{\sin\brp{\omega_o x} \fg(x)}
        &= \opLT\brs{-i^2\sin\brp{\omega_o x} \fg(x)}
      \\&= -i\opLT\brs{i\sin\brp{\omega_o x} \fg(x)}
        && \text{by \prope{homogeneous} property}
        && \text{\xref{thm:opLT_linear}}
      \\&= -i\opLT\brs{\sinh\brp{i\omega_o x} \fg(x)}
       %&&
        && \text{by \prefp{thm:sinh}}
      \\&=  -\frac{i}{2}\fG(s-i\omega_o) + \frac{i}{2}\fG(s+i\omega_o)
        && \text{by $\opLT\brs{\sin(\omega_o x)\fg(x)}$ result}
    \end{align*}

  \item Region of Convergence of $\opLT\brs{\cos\brp{\omega_o x} \fg(x)}$ and $\opLT\brs{\sin\brp{\omega_o x} \fg(x)}$:
    {\begin{align*}
      &\opROC\opLT\brs{\sfrac{\cos}{\sin}\brp{\omega_o x} \fg(x)}
      \\&= \opROC\opLT\brs{\brp{\frac{e^{ i\omega_o x}
                                  \pm e^{-i\omega_o x}}{2}} \fg(x)}
        &&  \text{by \thme{Euler's Identity}}
        &&  \text{\xref{thm:eid}}
      \\&= \opROC\brp{\opLT\brs{\frac{e^{ i\omega_o x}}{2} \fg(x)}
                  \pm \opLT\brs{\frac{e^{-i\omega_o x}}{2} \fg(x)}}
        &&  \text{by \prope{additive} property}
        &&  \text{\xref{thm:opLT_linear}}
      \\&= \opROC\opLT\brs{\brp{\frac{e^{-i\omega_o x}}{2}} \fg(x)} \seti
           \opROC\opLT\brs{\brp{\frac{e^{ i\omega_o x}}{2}} \fg(x)}
      \\&= \mathrlap{
           \intcc{A-\Real( i\omega)}{B-\Real( i\omega)} \seti
           \intcc{A-\Real(-i\omega)}{B-\Real(-i\omega)}
           }
      \\&= \intcc{A-0}{B-0} \seti
           \intcc{A-0}{B-0}
      \\&= \intcc{A}{B}
    \end{align*}}

  \item Region of Convergence of $\opLT\brs{\cosh\brp{\omega_o x} \fg(x)}$ and $\opLT\brs{\sinh\brp{\omega_o x} \fg(x)}$:
    {\begin{align*}
      &\opROC\opLT\brs{\sfrac{\cosh}{\sinh}\brp{\omega_o x} \fg(x)}
      \\&= \opROC\opLT\brs{\brp{\frac{e^{ \omega_o x}
                                  \pm e^{-\omega_o x}}{2}} \fg(x)}
        &&  \text{by def. $\cosh(x)$, $\sinh(x)$}
        &&  \text{\xref{def:cosh}}
      \\&= \opROC\brp{\opLT\brs{\frac{e^{ \omega_o x}}{2} \fg(x)}
                  \pm \opLT\brs{\frac{e^{-\omega_o x}}{2} \fg(x)}}
        &&  \text{by \prope{additive} property}
        &&  \text{\xref{thm:opLT_linear}}
      \\&= \opROC\opLT\brs{\brp{\frac{e^{-\omega_o x}}{2}} \fg(x)} \seti
           \opROC\opLT\brs{\brp{\frac{e^{ \omega_o x}}{2}} \fg(x)}
      \\&= \mathrlap{
           \intcc{A-\Real( \omega_o)}{B-\Real( \omega_o)} \seti
           \intcc{A-\Real(-\omega_o)}{B-\Real(-\omega_o)}
           }
      \\&= \brbl{\begin{array}{lM}
             \intcc{A+\Real( \omega_o)}{B-\Real( \omega_o)} & for $\omega\ge0$\\[2ex]
             \intcc{A-\Real( \omega_o)}{B+\Real( \omega_o)} & otherwise
           \end{array}}
      \\&= \intcc{A+\abs{\Real( \omega_o)}}{B-\abs{\Real( \omega_o)}}
        && \text{by definition of $\abs{x}$}
       %&& \text{\xref{def:abs}}
    \end{align*}}
\end{enumerate}
\end{proof}

\section{Causality properties}
%=======================================
%---------------------------------------
\begin{definition}
\footnote{
  \citerpu{betten2008H}{285}{https://link.springer.com/content/pdf/bbm:978-3-540-85051-9/1.pdf}
  }
\label{def:step}
%---------------------------------------
The \fnctd{Heaviside step function} $\step(x)$ or \fnctd{unit step function} is defined as
\defbox{
  \step(x) \eqd \brbl{\begin{array}{cM}
    1 & $\forall x\ge0$\\
    0 & otherwise
  \end{array}}
  }
\end{definition}

%---------------------------------------
\begin{theorem}
\footnote{
  \citerpgc{bracewell1978}{227}{007007013X}{Table 11.2 Some Laplace transforms}
  }
\label{thm:opLT_step}
%---------------------------------------
Let $\opLT$ be the \ope{Laplace Transform} operator \xref{def:opLT}
and $\step(x)$ the \fncte{unit step} function \xref{def:step}.
\thmbox{\begin{array}{>{\ds}lc>{\ds}r Dl C                 }
         \mc{3}{H}{Mapping}                    & \mc{2}{|H}{Region of Convergence} & \mc{1}{|H}{Domain}
  \\\hline
    \opLT\brs{\step(x)}  &=&  \frac{1}{s} & for & \Real(s) > 0 & \forall x\in\R
\end{array}}
\end{theorem}
\begin{proof}
\begin{align*}
  \opLT\brs{\step(x)}
    &\eqd \int_\R \step(x) e^{-sx}\dx
    &&
    && \text{by definition of $\opLT$}
    && \text{\xref{def:opLT}}
  \\&= \int_0^{\infty}  e^{-sx}\dx
    &&
    && \text{by definition of $\step(x)$}
    && \text{\xref{def:step}}
  \\&= \brlr{\frac{e^{-sx}}{-s}}_0^{\infty}
    &&
    && \text{by \thme{Fundamental Theorem of Calculus}}
  \\&= \lim_{x\to\infty}\brs{\frac{e^{-sx}}{-s}} - \brp{\frac{e^0}{-s}}
  \\&= 0 + \frac{1}{s}
    && \forall\Real(s) > 0
  \\&= \frac{1}{s}
    && \forall\Real(s) > 0
\end{align*}
\end{proof}

\begin{corollary}
\label{cor:opLT_step}
%---------------------------------------
Let $\opLT$ be the \ope{Laplace Transform} operator \xref{def:opLT}
and $\step(x)$ the \fncte{unit step} function \xref{def:step}.
\thmbox{\begin{array}{F >{\ds}lc>{\ds}r Dl C                 }
         & \mc{3}{H}{Mapping}                    & \mc{2}{|H}{Region of Convergence} & \mc{1}{|H}{Domain}
  \\\hline
    (1). & \opLT\brs{\step(-x) } &=& -\frac{1}{s}      & for & \Real(s) < 0 & \forall x\in\R
  \\(2). & \opLT\brs{\step(x-a)} &=& \frac{e^{-as}}{s} & for & \Real(s) < 0 & \forall a,x\in\R
  \\(2). & \opLT\brs{\step(a-x)} &=& -\frac{e^{as}}{s}  & for & \Real(s) > 0 & \forall a,x\in\R
\end{array}}
\end{corollary}
\begin{proof}
\begin{align*}
  \opLT\brs{\step(-x)}
    &= \opLT\brs{\step(x)}(-s)
    && \Real(s) < 0
    && \text{by \prope{reversal} property}
    && \text{\xref{cor:opLT_reversal}}
  \\&= \frac{-1}{s}
    && \Real(s) < 0
    && \text{by \prope{step} property}
    && \text{by \xref{thm:opLT_step}}
  \\\\
  \opLT\brs{\step(x-a)}
    &= e^{-as}\opLT\brs{\step(x)}
    && \Real(s) < 0
    && \text{by \prope{translation} property}
    && \text{\xref{thm:opLT_transversal}}
  \\&= \frac{e^{-as}}{s}
    && \Real(s) < 0
    && \text{by \prope{step} property}
    && \text{by \xref{thm:opLT_step}}
  \\\\
  \opLT\brs{\step(a-x)}
    &= e^{as}\opLT\brs{\step(-x)}
    && \Real(s) > 0
    && \text{by \prope{translation} property}
    && \text{\xref{thm:opLT_transversal}}
  \\&= -\frac{e^{as}}{s}
    && \Real(s) > 0
    && \text{by previous result}
\end{align*}
\end{proof}

\end{array}}
\end{corollary}

   %&& \text{by \prope{causality}}
   %&& \text{\xref{thm:opLT_step}}
  \\&= \frac{1}{s+\alpha}
    && \forall\Real(s) > -\Real(\alpha)
    &&
    &&
  \\\\
  \opLT\brs{e^{-\alpha x}\step(-x)}(s)
    &= \opLT\brs{\step(-x)}(s+\alpha)
    &&

    && \text{\xref{thm:opLT_mod}}

    && \mathrlap{\text{$\forall\Real(s) \in \intoo{-\infty-\Real(\alpha)}{0-(-\Real(\alpha))}$ by \prefp{cor:opLT_step}}}
   %&& \text{by \prope{anti-causality}}
   %&& \text{\xref{thm:opLT_step}}
  \\&= \frac{-1}{s+\alpha}
    && \forall\Real(s) < \Real(\alpha)
    && \text{by \prope{anti-causality}}
    && \text{\xref{cor:opLT_step}}
\end{align*}
\end{proof}

\begin{corollary}

\label{cor:opLT_stepcos}
%---------------------------------------
Let $\opLT$ be the \ope{Laplace Transform} operator \xref{def:opLT}
and $\step(x)$ the \fncte{unit step} function.
\corbox{\begin{array}{F>{\ds}lc>{\ds}c                                                     D     l                             C }
    &\mc{3}{H}{Mapping}                                                                  & \mc{2}{|H}{Region of Convergence} & \mc{1}{|H}{Domain}
  \\\hline
         (1).& \opLT\brs{\cos(\omega_o x)\step( x)} &=& \frac{s}{s^2+\omega_o^2}         & for & \Real(s) > 0 & x,\omega_o\in\R
  \\[2ex](2).& \opLT\brs{\sin(\omega_o x)\step( x)} &=& \frac{\omega_o}{s^2+\omega_o^2}  & for & \Real(s) > 0 & x,\omega_o\in\R
  \\[2ex](3).& \opLT\brs{\cos(\omega_o x)\step(-x)} &=& \frac{-s}{s^2+\omega_o^2}        & for & \Real(s) < 0 & x,\omega_o\in\R
  \\[2ex](4).& \opLT\brs{\sin(\omega_o x)\step(-x)} &=& \frac{-\omega_o}{s^2+\omega_o^2} & for & \Real(s) < 0 & x,\omega_o\in\R
\end{array}}
\end{corollary}
\begin{proof}
\begin{align*}
  \opLT\brs{\cos(\omega_o x)\step(x)}(s)
    &= \frac{1}{2}\opLT\brs{\step(x)}(s-i\omega_o)
     + \frac{1}{2}\opLT\brs{\step(x)}(s+i\omega_o)
    &&
    && \text{by \prope{modulation}}
    && \text{\xref{cor:opLT_mod}}
  \\&= \frac{1}{2}\brs{\frac{1}{s-i\omega_o}}
     + \frac{1}{2}\brs{\frac{1}{s+i\omega_o}}
    && \Real(s) > 0
    && \text{by \prope{causal} prop.}
    && \text{\xref{thm:opLT_step}}
  \\&= \mathrlap{
       \frac{1}{2}\brs{\frac{1}{s-i\omega_o}}\brs{\frac{s+i\omega_o}{s+i\omega_o}}
     + \frac{1}{2}\brs{\frac{1}{s+i\omega_o}}\brs{\frac{s-i\omega_o}{s-i\omega_o}}
     \qquad\text{(\ope{Rationalizing the Denominator})}
     }
  \\&= \frac{1}{2}\brs{\frac{(s+i\omega_o) + (s-i\omega_o)}{s^2+\omega_o^2}}
    && \Real(s) > 0
  \\&= \frac{s}{s^2+\omega_o^2}
    && \Real(s) > 0
  \\
  \opLT\brs{\sin(\omega_o x)\step(x)}(s)
    &= -\frac{i}{2}\opLT\brs{\step(x)}(s-i\omega_o)
       +\frac{i}{2}\opLT\brs{\step(x)}(s+i\omega_o)
    &&
    && \text{by \prope{modulation}}
    && \text{\xref{cor:opLT_mod}}
  \\&= -\frac{i}{2}\brs{\frac{1}{s-i\omega_o}}
       +\frac{i}{2}\brs{\frac{1}{s+i\omega_o}}
    && \Real(s) > 0
    && \text{by \prope{causal} prop.}
    && \text{\xref{thm:opLT_step}}
  \\&= \mathrlap{
     - \frac{i}{2}\brs{\frac{1}{s-i\omega_o}}\brs{\frac{s+i\omega_o}{s+i\omega_o}}
     + \frac{i}{2}\brs{\frac{1}{s+i\omega_o}}\brs{\frac{s-i\omega_o}{s-i\omega_o}}
     \qquad\text{(\ope{Rationalizing the Denominator})}
     }
  \\&= \frac{i}{2}\brs{\frac{-(s+i\omega_o) + (s-i\omega_o)}{s^2+\omega_o^2}}
    && \Real(s) > 0
  \\&= \frac{\omega_o}{s^2+\omega_o^2}
    && \Real(s) > 0
\end{align*}

\begin{align*}
  \opLT\brs{\step(-x)\cos(\omega_o x)}(s)
    &= \opLT\brs{\step(-x)\cos(\omega_o(-x))}(s)
    &&
    && \text{by \prope{even} property of $\cos(x)$}
    && \text{\xref{thm:cos-x}}
  \\&= \opLT\brs{\step(x)\cos(\omega_o x)}(-s)
    &&
    && \text{by \prope{reversal} property}
    && \text{\xref{cor:opLT_reversal}}
  \\&= \frac{(-s)}{(-s)^2+\omega_o^2}
    && \Real(s) < 0
    && \text{by (1)}
    &&
  \\&= \frac{-s}{s^2+\omega_o^2}
    && \Real(s) < 0
  \\
  \opLT\brs{\sin(\omega_o x)\step(-x)}(s)
    &= \opLT\brs{-\sin(\omega_o(-x))\step(-x)}(s)
    &&
    && \text{by \prope{odd} property of $\sin(x)$}
    && \text{\xref{thm:sin-x}}
  \\&= -\opLT\brs{\sin(\omega_o(-x))\step(-x)}(s)
    &&
    && \text{by \prope{homogeneous} property}
    && \text{\xref{thm:opLT_linear}}
  \\&= -\opLT\brs{\sin(\omega_o x)\step(x)}(-s)
    &&
    && \text{by \prope{reversal} property}
    && \text{\xref{cor:opLT_reversal}}
  \\&= -\brs{\frac{\omega_o}{(-s)^2+\omega_o^2}}
    && \Real(s) < 0
    && \text{by (2)}
    &&
  \\&= \frac{-\omega_o}{s^2+\omega_o^2}
    && \Real(s) < 0
\end{align*}
\end{proof}

        && \text{by \prope{modulation}}
        && \text{\xref{cor:opLT_mod}}

        &= \intcc{A+\abs{\Real(\omega_o)}}{B-\abs{\Real(\omega_o)}}
        && \text{by \prefp{cor:opLT_mod}}
      \\&= \intoo{-\infty+\abs{\Real(\omega_o)}}{0-\abs{\Real(\omega_o)}}

      \\&\implies \opROC\opLT\brs{\cosh(\omega_o x)\step(-x)} < \abs{\Real(\omega_o)}
    \end{align*}
\end{enumerate}

  \\\opLT\brs{\sin(\omega_o x)} & is \propb{divergent} & \forall s\in\C & \forall x,\omega_o\in\R\setd\setn{0}
\end{array}}
\end{corollary}
\begin{proof}
\begin{align*}
  \opLT\brs{\cos(\omega_o x)}
    &= \mcom{\opLT\brs{\step( x)\cos(\omega_o x)}}{$\forall \Real(s)>0$}
     + \mcom{\opLT\brs{\step(-x)\cos(\omega_o x)}}{$\forall \Real(s)<0$}
    && \text{by \prefp{cor:opLT_stepcos}}
  \\&= \mcom{\frac{s}{s^2+\omega_o^2}}{$\forall \Real(s)>0$}
     + \mcom{\frac{-s}{s^2+\omega_o^2}}{$\forall \Real(s)<0$}
    && \text{by \prefp{cor:opLT_stepcos}}
  \\&= \brbl{\begin{array}{cl}
         0      & \forall\Real(s)\in\intoo{-\infty}{0}\seti\intoo{0}{\infty}=\emptyset\\
         \infty & \forall s\in\C
       \end{array}}
  \\&\text{$\implies$ $\opLT\brs{\cos(\omega_o x)}$ is \propb{divergent} $\forall s\in\C$}
  \\
  \opLT\brs{\sin(\omega_o x)}
    &= \mcom{\opLT\brs{\step( x)\sin(\omega_o x)}}{$\forall \Real(s)>0$}
     + \mcom{\opLT\brs{\step(-x)\sin(\omega_o x)}}{$\forall \Real(s)<0$}
    && \text{by \prefp{cor:opLT_stepcos}}
  \\&= \mcom{\frac{\omega_o}{s^2+\omega_o^2}}{$\forall \Real(s)>0$}
     + \mcom{\frac{-\omega_o}{s^2+\omega_o^2}}{$\forall \Real(s)<0$}
    && \text{by \prefp{cor:opLT_stepcos}}
  \\&= \brbl{\begin{array}{cl}
         0      & \forall\Real(s)\in\intoo{-\infty}{0}\seti\intoo{0}{\infty}=\emptyset\\
         \infty & \forall s\in\C
       \end{array}}
  \\&\text{$\implies$ $\opLT\brs{\sin(\omega_o x)}$ is \propb{divergent} $\forall s\in\C$}
\end{align*}
\end{proof}

      && \text{\xref{def:step}}

      && \forall\Real(s)\in\intoo{-\Real(\alpha)}{\Real(\alpha)}

\end{proof}

    &\mc{3}{H}{Mapping}                                                                                            & \mc{2}{|H}{Region of Convergence}& \mc{1}{|H}{Domain}
  \\\hline%                                                                                                        &                                  &
    (1).& \opLT\brs{\sin(\omega_o x)e^{-\alpha\abs{x}}\step( x)}  &=& \frac{ \omega_o}{(s+\alpha)^2 + \omega_o^2}  & for & \Real(s) \in \intoo{-A}{A} & x,\alpha\in\R
  \\(2).& \opLT\brs{\sin(\omega_o x)e^{-\alpha\abs{x}}\step(-x)}  &=& \frac{-\omega_o}{(s-\alpha)^2 + \omega_o^2}  & for & \Real(s) \in \intoo{-A}{A} & x,\alpha\in\R
  \\(3).& \opLT\brs{\sin(\omega_o x)e^{-\alpha\abs{x}}         }  &=& \frac{ \omega_o}{(s+\alpha)^2 + \omega_o^2} +  \frac{-\omega_o}{(s-\alpha)^2 + \omega_o^2}  & for & \Real(s) \in \intoo{-A}{A} & x,\alpha\in\R
\end{array}}
\end{corollary}
\begin{proof}
\begin{enumerate}
  \item Proof for (1):
    \begin{align*}
      &\opLT\brs{\sin(\omega_o x)e^{-\alpha\abs{x}}\step(x)}
      \\&= \opLT\brs{\sin( \omega_o x)e^{-\alpha x}\step( x)}
        && \text{by definition of $\step(x)$}
        && \text{\xref{def:step}}
      \\&= \frac{s+\alpha}{(\omega_o)^2 + \omega_o^2}
        && \forall \Real(s) \in \intoo{-\Real(\alpha)}{\Real(\alpha)}
        && \text{by \prefp{cor:opLT_stepcos} and \prefp{thm:opLT_mod}}
    \end{align*}

  \item Proof for (2):
    \begin{align*}
      &\opLT\brs{\sin(\omega_o x)e^{-\alpha\abs{x}}\step(-x)}
      \\&= \opLT\brs{\sin(-\omega_o x)e^{ \alpha x}\step(-x)}
        && \text{by definition of $\step(x)$}
        && \text{\xref{def:step}}
      \\&= \opLT\brs{-\sin(\omega_o x)e^{ \alpha x}\step(-x)}
        && \text{by \prope{odd} property of $\sin(x)$}
        && \text{\xref{thm:sin-x}}
      \\&= -\opLT\brs{\sin(\omega_o x)e^{ \alpha x}\step(-x)}
        && \text{by \prope{homogeneous} property}
        && \text{\xref{thm:opLT_linear}}
      \\&= \frac{-\omega_o}{(s-\alpha)^2 + \omega_o^2}
        && \forall \Real(s) \in \intoo{-\Real(\alpha)}{\Real(\alpha)}
        && \text{by \prefp{thm:opLT_mod} and \pref{cor:opLT_stepcos}}
    \end{align*}

  \item Proof for (3):
    \begin{align*}
      \opLT\brs{\sin(\omega_o x)e^{-\alpha\abs{x}}}
        &= \opLT\brs{\sin(\omega_o x)e^{-\alpha\abs{x}}\step(x)}
         + \opLT\brs{\sin(\omega_o x)e^{-\alpha\abs{x}}\step(-x)}
      \\&= \opLT\brs{\sin( \omega_o x)e^{-\alpha x}\step( x)}
         + \opLT\brs{\sin(-\omega_o x)e^{ \alpha x}\step(-x)}
      \\&= \frac{\omega_o}{(s+\alpha)^2 + \omega_o^2} +  \frac{-\omega_o}{(s-\alpha)^2 + \omega_o^2}
        && \forall \Real(s) \in \intoo{-\Real(\alpha)}{\Real(\alpha)}
    \end{align*}
%\end{proof}

\pref{thm:opLT_conv} (next) demonstrates that multiplication in the ``time domain"
is equivalent to convolution in the ``s domain" and
vice-versa.

%--------------------------------------
\begin{theorem}[\thmd{convolution theorem}]
\footnote{
  \citerpg{bracewell1978}{224}{007007013X},
  \citerppg{bachman2002}{268}{270}{9780387988993},
  \citerpg{bachman1964}{8}{9781483267562}
  }
\label{thm:opLT_conv}
%--------------------------------------
Let $\opLT$ be the \ope{Laplace Transform} operator \xref{def:opLT}
and $\conv$ the convolution operator \xref{def:conv}.
Let $A$, $B$, $C$, and $D$ be defined as in \prefpp{cor:opLT_linear}.
\thmbox{
\begin{array}{rcllC}
  \opLT\brs{\ff(x)\conv\fg(x)}(s)
    &=& \brs{\opLT\ff}(s)\, \brs{\opLT\fg}(s)
    & \forall \Real(s)\in\intcc{A}{B}\seti\intcc{C}{D}
    & \forall\ff,\fg\in\spLLRBu
  \\
  \opLT\brs{\ff(x) \fg(x)}(s)
    &=& \Lscale\brs{\opLT\ff}(s) \conv \brs{\opLT\fg}(s)
    & \forall \Real(s)\in\intcc{A+C}{B+D},\,c\in\intoo{A}{B}
    & \forall\ff,\fg\in\spLLRBu.
\end{array}}
\end{theorem}
\begin{proof}
\begin{align*}
   \opLT \brs{\ff(x)\conv\fg(x)}(s)
     &= \opLT \brs{ \int_{u\in\R} \ff(u)\fg(x-u) \du }(s)
     && \text{by definition of $\conv$} &&\text{\xref{def:conv}}
   \\&=  \int_{u\in\R} \ff(u)\brs{\opLT \fg(x-u)}(s) \du
   \\&=  \int_{u\in\R} \ff(u) e^{-su} \; \brs{\opLT\fg(x)}(s) \du
     && \text{by \prope{translation} property}
     && \text{\xref{thm:opLT_shift}}
   \\&= \mcom{\brp{\Lscale\int_{u\in\R} \ff(u) e^{-su} \du}}
             {$\brs{\opLT\ff}(s)$                          } \;
        \brs{\opLT\fg}(s)
   \\&= \brs{\opLT\ff}(s)\;  \brs{\opLT\fg}(s)
     && \Real(s)\in\intcc{A}{B}\seti\intcc{C}{D}
     && \text{by definition of $\opLT$}
   \\
   \\
   \opLT[\ff(x)\fg(x)](s)
     &= \opLT\brs{\brp{\opLT^{-1}\opLT\ff(x)}\;\fg(x)}(s)
     && \text{by def. of operator inverse} &&\text{\ifxref{operator}{def:op_inv}}
   \\&= \opLT\brs{\brp{ \Lscalei \int_{v\in\R} \brs{\opLT\ff(x)}(v) e^{sxv} \dv }\;\fg(x)}(s)
     && \text{by \prefp{thm:opLTi}}
   \\&= \Lscalei \int_{v\in\R} \brs{\opLT\ff(x)}(v) \brs{\opLT\brp{e^{sxv} \;\fg(x)}}(s,v) \dv
   \\&= \Lscalei \int_{v\in\R} \brs{\opLT\ff(x)}(v) \brs{\opLT\fg(x)}(s-v) \dv
     && \text{by \prefp{thm:opLT_shift}}
   \\&= \Lscalei\brs{\opLT\ff}(s)\conv \brs{\opLT\fg}(s)
     && \text{by definition of $\conv$} &&\text{\xref{def:conv}}
\end{align*}
\end{proof}

\section{Calculus properties}

\begin{theorem}
\footnote{
  \citerpcu{betten2008L}{301}{(B.27)}{https://link.springer.com/content/pdf/bbm:978-3-540-85051-9/1.pdf},
  \citerpcu{levy1958}{15}{Equation 0.7}{https://archive.org/details/c_cowles_ieee_Levy/page/n15/}
  }
\label{thm:opLT_diff}
\label{thm:opLT_int}
%---------------------------------------
Let $\opLT$ be the \ope{Laplace Transform} operator \xref{def:opLT}.
\thmbox{\begin{array}{>{\ds}rc>{\ds}l}
    \opLT \brs{\ddx \fg(x)                      }(s) &=& s           \brs{\opLT\fg}(s) + \lim_{x\to\infty}\brs{e^{-sx}\fg(x) -  e^{sx}\fg(-x)}
  \\\opLT \brs{\int_{u=-\infty}^{u=x} \fg(u) \du}(s) &=& \frac{1}{s} \brs{\opLT\fg}(s)
\end{array}}
\end{theorem}
\begin{proof}
  \begin{align*}
    \opLT \brs{\ddx \fg(x)}
      &\eqd \int_{x\in\R} \mcom{\brs{\ddx \fg(x)}}{$\dv$} \mcom{e^{-sx}}{$u$} \dx
      && \text{by definition of $\opLT$}
      && \text{\xref{def:opLT}}
    \\&= \brs{\mcom{e^{-sx}}{$u$} \mcom{\fg(x)}{$v$}}_{x=-\infty}^{x=+\infty}
        -\brs{\int_{x\in\R} \mcom{\fg(x)}{$v$} \mcom{(-s)e^{-s x}}{$\du$} \dx}
      && \text{by \thme{Integration by Parts}}
    \\&= \mathrlap{ \lim_{x\to\infty}\brs{e^{-sx}\fg(x)}
        - \lim_{x\to-\infty}\brs{e^{-sx}\fg(-\infty)}
        + s\int_{x\in\R} \fg(x) e^{-sx} \dx}
    \\&=  s \brs{\opLT\fg}(s)
        + \lim_{x\to\infty}\brs{e^{-sx}\fg(x) - e^{sx}\fg(-x)}
      && \text{by definition of $\opLT$}
      && \text{\xref{def:opLT}}
  \end{align*}

  \begin{align*}
    \opLT \int_{v=-\infty}^{v=x} \fg(v) \dv
      &\eqd \int_{x=-\infty}^{x=+\infty} \brs{\int_{v=-\infty}^{v=x} \fg(v) \dv} e^{-s x} \dx
      && \text{by definition of $\opLT$}
      && \text{\xref{def:opLT}}
    \\&= \int_{x=-\infty}^{x=+\infty} 
         \brs{\int_{v=-\infty}^{v=+\infty} \fg(v) \step(x-v) \dv} 
          e^{-s x} \dx
      && \text{by definition of $\step(x)$}
      && \text{\xref{def:step}}
    \\&= \int_{v\in\R} \fg(v) \brs{\int_{x\in\R} \step(x-v) e^{-s x} \dx} \dv
    \\&= \int_{v\in\R} \fg(v) \brs{e^{-vs}\frac{1}{s}} \dv
      && \forall \Real(s)<0
      && \text{by \prefp{cor:opLT_step}}
    \\&= \frac{1}{s}\int_{v\in\R} \fg(v) e^{-vs} \dv
      && \forall \Real(s)<0
      && \text{by \prefp{cor:opLT_step}}
    \\&\eqd \frac{1}{s} \brs{\opLT\fg}(s)
      && \text{by definition of $\opLT$}
      && \text{\xref{def:opLT}}
  \end{align*}
\end{proof}

\begin{example}
\label{ex:opLT_capZ}
%--------------------------------------
\exboxt{
  The \ope{Laplace Transform} of the \prope{impedance} $Z(s)$ of an ideal capacitor with \prope{capacitance} C,
  \\ initial voltage $\fv(-\infty)=0$, and finite total energy stored over all time, is
  \\\indentx$\ds Z(s) = \frac{1}{sC}$.
  }
\end{example}
\begin{proof}
\begin{enumerate}
  \item An ideal capacitor is modeled with the equation $\ds\fnct{i}(t) = C\ddt\fv(t)$.
  \item lemma: Because initial voltage is $0$, $\ds\lim_{t\to\infty}\brs{e^{st}\fv(-t)}=0$. \label{ilem:opLT_capZ1}
  \item lemma: Because the capacitor has finite energy, $\ds\lim_{t\to\infty}\brs{e^{-st}\fv(t)}=0$. \label{ilem:opLT_capZ2}
  \item
    \begin{align*}
      \fZ(s)
        &  \eqd \frac{\fV(s)}
                     {\fI(s)}
        && \text{by definition of \prope{impedance} $\fZ(s)$}
      \\&  \eqd \frac{\fV(s)}
                     {\ds\opLT \brs{\fnct{i}(t)} }
        && \text{by definition of \ope{Laplace Transform} $\fI(s)$ of $\fnct{i}(t)$}
      \\&  \eqd \frac{\fV(s)}
                     {\ds\opLT \brs{ C \ddt\fv(t)} }
        && \text{by definition of \ope{ideal Capacitor} with \prope{Capacitance} $C$}
      \\&  =    \frac{\fV(s)}
                     {\ds C\opLT \brs{ \ddt\fv(t)} }
        && \text{by \prope{homogeneous} property of $\opLT$ \prefpp{thm:opLT_linear}}
      \\&  =    \frac{\fV(s)}
                     {\ds Cs\brp{\opLT \brs{ \fv(t)} + \lim_{t\to\infty}\brs{e^{-st}\fv(t) + e^{st}\fv(-t)} }}
        && \text{by \prefp{thm:opLT_diff}}
      \\&  =    \frac{\fV(s)}
                     {\ds Cs\opLT \brs{ \fv(t)}}
        && \text{by \pref{ilem:opLT_capZ1} and \pref{ilem:opLT_capZ2}}
      \\&  \eqd \frac{\fV(s)}
                     {\ds Cs\opLT \fV(s)}
        && \text{by definition of $\fV(s)$}
      \\&  = \frac{1}{sC}
    \end{align*}
\end{enumerate}
\end{proof}

\section{Moment properties}

%--------------------------------------
\begin{definition}
\footnote{
  \citePpp{jawerth}{16}{17},
  \citePp{sweldens93}{2},
  \citerpg{vidakovic}{83}{9780471293651}
  %\citerp{goswami}{102}    \\
  %\citerpp{mallat}{241}{243}
  }
\label{def:moment}
%--------------------------------------
\defboxt{
  The quantity $\fM_n$ is the \vald{$n$th moment} of a function $\ff(x)\in\spLLR$ if
  \\\indentx$\ds \fM_n \eqd \int_{\R} x^n \ff(x) \dx \qquad \text{for }n\in\Znn$.
  }
\end{definition}

%--------------------------------------
\begin{theorem}
\footnote{
  \citerppg{goswami}{38}{39}{9780471197485}
  }
\label{thm:opLT_moment}
%--------------------------------------
Let $\fM_n$ be the \fncte{$n$th moment} \xref{def:moment} of a function $\ff(x)\in\spLLR$.
\thmbox{\begin{array}{>{\ds}rc>{\ds}l@{\qquad}C}
  \fM_n  \eqd \int_\R x^n \ff(x) \dx &=&  \brs{(-1)^{n}\brs{\opddsn\opLT\ff}(s)}_{s=0} & \forall n\in\Znn,\,\ff\in\spLLR\\
\end{array}}
\end{theorem}
\begin{proof}
  \begin{align*}
    (-1)^n\brs{\opddsn \opLT\ff}(s)_{s=0}
      &= (-1)^n\brs{\opddsn \int_\R \ff(x) e^{-sx} \dx}_{s=0}
      && \text{by definition of $\opLT$}
      && \text{\xref{def:opLT}}
    \\&= (-1)^n\brs{\int_\R \ff(x) \brs{\opddsn e^{-sx}}\dx}_{s=0}
    \\&= (-1)^n\brs{\int_\R \ff(x) (-x)^n e^{-sx}\dx}_{s=0}
    \\&= (-1)^n\int_\R \ff(x) (-x)^n \dx
    \\&= (-1)^{2n}\int_\R \ff(x) x^n \dx
    \\&\eqd \fM_n
      && \text{by definition of $\fM_n$}
      && \text{\xref{def:Mn_ft}}
  \end{align*}
\end{proof}

%--------------------------------------
\begin{corollary}
\footnote{
  \citerppg{vidakovic}{82}{83}{9780471293651},
  \citerpp{mallat}{241}{242}
  }
%--------------------------------------
Let $\fM_n$ be the \fncte{$n$th moment} \xref{def:moment} of a function $\ff(x)$ in $\spLLR$.
\corbox{
  \fM_n=0
  \qquad\iff\qquad
  \brs{\opddsn\opLT\ff}_{s=0} =0
  \qquad\scy\forall n\in\Znn
  }
\end{corollary}
\begin{proof}
\begin{enumerate}
  \item Proof for $(\implies)$ case:
    \begin{align*}
      0 &= (-1)^n\cdot0
      \\&= (-1)^n\fM_n
        && \text{by left hypothesis}
      \\&= (-1)^n\int_\R x^n \ff(x) \dx
      \\&= (-1)^n\brs{(-1)^n \opddsn\opLT\ff}_{s=0}
        && \text{by \prefp{thm:opLT_moment}}
      \\&= \brs{\opddsn\opLT\ff}_{s=0}
    \end{align*}

  \item Proof for $(\impliedby)$ case:
    \begin{align*}
      0 &= \brs{\opddsn\opLT\ff}_{s=0}
        && \text{by right hypothesis}
      \\&= \brs{\opddsn\int_\R \ff(x) e^{-sx}\dx}_{s=0}
        && \text{by definition of $\opLT$}
        && \text{\xref{def:opLT}}
      \\&= \brs{\int_\R \ff(x) \opddsn e^{-sx}\dx}_{s=0}
      \\&= \brs{\int_\R \ff(x) \brs{(-1)^n x^n e^{-sx}}\dx}_{s=0}
      \\&= (-1)^n \int_\R \ff(x) x^n \dx
    \end{align*}
\end{enumerate}
\end{proof}
