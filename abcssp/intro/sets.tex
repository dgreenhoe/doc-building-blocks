\subsection{Sets}
%=======================================
%---------------------------------------
\begin{definition}
\label{def:emptyset}
\footnote{
  \citerpg{halmos1960}{8}{0387900926},
  \citerpg{kelley1955}{3}{0387901256},
  \citerp{kuratowski_stt}{26}
  }
%---------------------------------------
Let $\setX$ be a \structe{set}.\footnotemark
\defboxt{\indxs{\emptyset}
  The \structd{empty set} $\emptyset$ is defined as \quad
  $\emptyset \eqd \set{x\in\setX}{x\ne x}$.
  }
\end{definition}
\footnotetext{
The mathematical structure called \structe{set} is left undefined in this paper.
  For more information on \structe{set}s, see for example
  \citePpp{zermelo1908}{263}{267} (7 axioms),
  \citerc{zermelo1908e}{(English translation of previous reference)},
  \citeP{fraenkel1922},
  \citerppgc{halmos1960}{1}{6}{0387900926}{Naive set theory},
  \citerp{wolf}{139}
  }


%---------------------------------------
\begin{definition}
\footnote{
  Notation $\R, \Znn, \Zp$, etc.:
  \hie{Bourbaki notation}.
  References: 
  \citerpg{davis2005}{9}{0071243399},
  \citerpg{cohn2012}{3}{9781447104759}
  }
\label{def:R}
\label{def:Rnn}
\label{def:Rp}
\label{def:Rx}
\label{def:Z}
%---------------------------------------
Let $\R$ be the \structd{set of real numbers}.
Let $\Rnn\eqd\set{x\in\R}{x\ge0}$ be the \structd{set of non-negative real numbers}.
Let $\Rp\eqd\set{x\in\R}{x>0}$ be the \structd{set of postive real numbers}.
Let $\Rx\eqd\R\setu\setn{-\infty,\infty}$ be the set of \structd{extended real numbers}.
\footnote{
  \citerppgc{rana2002}{385}{388}{0821829742}{Appendix A}
  }
Let $\Z$ be the \structd{set of integers}.
Let $\Znn\eqd\set{n\in\Z}{n\ge0}$ be the \structd{set of whole numbers}.
Let $\Zp\eqd\set{n\in\Z}{n\ge1}$ be the \structd{set of natural numbers}.
Let $\Zext\eqd\Z\setu\setn{-\infty,\infty}$ be the \structe{extended set of integers}.
\end{definition}
