\section{Introduction}
%=======================================
The literature offers varying and in general incompatible definitions of the
\fncte{cross-correlation} function $\Rxy(n,m)$ and its \ope{jointly wide-sense stationary} \ref{def:jwss} 
special case $\Rxy(m)$.
The choice of definition has consequences for results involving the \fncte{cross-spectral density} function
$\Swxy(\omega)$, 
which is defined as the \ope{Discrete-time Fourier Transform} of $\Rxy(m)$,
as well as the more general $\Szxy(z)$,
which is defined as the \ope{Z-Transform} of $\Rxy(m)$:
%---------------------------------------
\begin{definition}
%\footnote{

%  }
\label{def:Swxy}
\label{def:Szxy}
\label{def:dtft}
\label{def:jwss}
%---------------------------------------
\defbox{
  \Szxy(z)       \eqd \mcom{\opZ\Rxy(m) \eqd \sum_{m\in\Z}\Rxy(m) z^{-m}}
                           {\opd{Z-Transform} of $\Rxy(m)$}
  \qquad\qquad
  \Swxy(\omega)  \eqd \mcom{\opDTFT\Rxy(m) \eqd \sum_{m\in\Z}\Rxy(m) e^{-i\omega m}}
                           {\opd{Discrete-time Fourier Transform} of $\Rxy(m)$}
  }
\end{definition}
