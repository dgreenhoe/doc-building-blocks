%---------------------------------------
\begin{definition}
%---------------------------------------
Let $(X,\le)$ be an (partially or totally) ordered set and $A\subseteq X$.

A {\bf lower bound} is any element in the set
$\opL A=\set{x\in X}{x\le a \; \forall a\in A}$.

An {\bf upper bound} is any element in the set
$\opU A=\set{x\in X}{a\le x \; \forall a\in A}$.

A {\bf greatest lower bound} is any element in the set 
$\bigwedge A = \set{x\in \opL A}{y\le x\; \forall y\in\opL A}$.

A {\bf least upper bound} is any element in the set 
$\bigvee A = \set{x\in \opU A}{x\le y\; \forall y\in\opU A}$.
\end{definition}

%---------------------------------------
\begin{definition}
%---------------------------------------
\defbox{\begin{array}{lll}
  \mc{3}{l}{\text{
    A {\bf lattice} is any tupple $(X,\le,\join,\meet)$ that satisfies
    }} \\
  & 1. & (X,\le) \text{ is a (partially or totally) ordered set}  \\
  & 2. & \exists\quad x\join y \text{ (least upper bound) } \forall x,y\in X \\
  & 3. & \exists\quad x\meet y \text{ (greatest lower bound) } \forall x,y\in X
\end{array}}
\end{definition}

\begin{example}
\label{ex:lat_abc}
%---------------------------------------
Let $A,B\subseteq\setn{a,b,c}$.
The partially ordered set $(\powerset(\setn{a,b,c}),\subseteq)$ 
(see Figure~\ref{fig:lat_hasse_abc}) is a lattice with
\exbox{\begin{array}{lll @{\qquad}C @{\qquad}D}
  A\join   B &=& A\cup B   
    & \forall A,B\in\powerset\left(\setn{a,b,c}\right) 
    & (least upper bound)  
    \\
  A\meet B &=& A\cap B   
    & \forall A,B\in\powerset\left(\setn{a,b,c}\right) 
    & (greatest lower bound)
\end{array}}
\end{example}

\begin{example}
%---------------------------------------
Let $X\eqd\set{n\in\Z}{n=5^a \cdot 3^b \cdot 2^c,\;a,b,c\in\setn{0,1}}$.
Let $n|m$ denote that $m$ divides $n$.
The partially ordered set $(X,|)$ 
(see Figure~\ref{fig:lat_hasse_532}) is a lattice with
\exbox{\begin{array}{lll @{\qquad}C @{\qquad}D}
  n \join m &=& \lcm(n,m)   
    & \forall n,m\in X
    & (least upper bound)  
    \\
  n \meet m &=& \gcd(n,m)
    & \forall n,m\in X
    & (greatest lower bound)
\end{array}}
\end{example}

\begin{example}
\citep{burris}{11}
%---------------------------------------
Let $(X,\tau)$ be any topological space.
\exbox{\begin{array}{lll @{\qquad}C @{\qquad}D}
  \mc{3}{l}{\text{$(\tau,\subseteq)$ is a lattice.}}
  \\
  A\join   B &=& A\cup B   \in \tau
    & \forall A,B\in\tau
    & (least upper bound)  
    \\
  A\meet B &=& A\cap B \in \tau  
    & \forall A,B\in\tau
    & (greatest lower bound)
\end{array}}
\end{example}

\begin{theorem}
\label{thm:lattice}
\citepp{burris}{5}{8}
%\citep{johnstone}{1}
\index{commutative      }
\index{associative      }
\index{idempotent       }
\index{absorptive}
\index{contractive}
%\index{identity}
%---------------------------------------
Let $(X,\le,\join,\meet)$ be a lattice.
For all $x,y,z\in X$
\formbox{
\renewcommand{\arraystretch}{1.25}
\begin{array}{rcl @{\qquad} rcl @{\qquad}D}
      x \join x &=& x
    & x \meet x &=& x
    & (idempotent)
    \\ 
      x \join y &=& y \join x     
    & x \meet y &=& y \meet x  
    & (commutative)
    \\ 
      x \join (x \meet y) &=& x
    & x \meet (x \join y) &=& x
    & (absorptive/contractive)  
    \\ 
      (x \join y) \join z &=& x \join (y \join z)
    & (x \meet y) \meet z &=& x \meet (y \meet z)
    & (associative)
%   \\
%      x \join 0 &=& x
%    & x \meet 1 &=& x
%    & (identity)  
\end{array}}
\end{theorem}



%---------------------------------------
\begin{theorem}[Distributive inequalities]
\label{thm:lat_dis_<}
\citepp{burris}{12}{13}
\index{distributive}
%---------------------------------------
Let $(X,\le,\join,\meet)$ be a lattice.
\formbox{\begin{array}{rcl}
  (x \meet y) \join (x \meet z) &\le& x \meet (y \join z)     \\
  x \join (y \meet z)           &\le& (x \join y) \meet (x \join z)  
\end{array}}
\end{theorem}
\begin{proof}
\begin{align*}
  (x \meet y) \join (x \meet z)
    &\le \mcom{\big[ x\meet(y\join z)\big]}{$x\meet y\le \; \uparrow$} \join 
         \mcom{\big[ x\meet(y\join z)\big]}{$x\meet z\le \; \uparrow$}
  \\&=   x\meet(y\join z)
    &&   \text{by Theorem~\ref{thm:lattice} (idempotent property)}
\\
\\
  x\join(y\meet z)
    &=   \mcom{\big[ x\join(y\meet z)\big]}{$\uparrow\;\le x\join y$} \meet 
         \mcom{\big[ x\join(y\meet z)\big]}{$\uparrow\;\le x\join z$}
    &&   \text{by Theorem~\ref{thm:lattice} (idempotent property)}
  \\&\le (x \join y) \meet (x \join z)
\end{align*}
\end{proof}


%---------------------------------------
\begin{theorem}[Modular inequality]
\label{thm:lat_mod_<}
\citep{burris}{13}
\index{modular}
%---------------------------------------
Let $(X,\le,\join,\meet)$ be a lattice.
\formbox{
  x\le y
  \qquad\implies\qquad
  x\join(y\meet z) \le y\meet(x\join z)
}
\end{theorem}
\begin{proof}
\begin{align*}
  x\join(y\meet z)
    &=   \mcom{\big[ x\join(y\meet z)\big]}{$\uparrow\;\le x\join y$} \meet
         \mcom{\big[ x\join(y\meet z)\big]}{$\uparrow\;\le x\join z$}
    &&   \text{by Theorem~\ref{thm:lattice} (idempotent property)}
  \\&\le (x\join y) \meet (x\join z)
  \\&\le y \meet (x\join z)
    &&   \text{by left hypothesis}
\end{align*}
\end{proof}

%---------------------------------------
\begin{theorem}
\label{thm:lat_dis}
\citep{burris}{12}
\index{distributive}
%---------------------------------------
Let $(X,\le,\join,\meet)$ be a lattice.
\formbox{
  \mcom{x \meet (y \join z) = (x\meet y) \join (x\meet z)}{$\meet$ distributes over $\join$}
  \iff
  \mcom{x \join (y \meet z) = (x\join y) \meet (x\join z)}{$\join$ distributes over $\meet$}
  }
\end{theorem}
\begin{proof}
\begin{align*}
\intertext{1. Proof that (left hyothesis) $\implies$   (right hypothesis):}
  x \join (y \meet z)
    &= \mcom{\big[ x \join (x\meet y)\big]}{expand $x$ wrt $y$} \join (y \meet z)
    && \text{by Theorem~\ref{thm:lattice}---absorptive property}
  \\&= x \join \big[ (x\meet y) \join (y \meet z) \big]
    && \text{by Theorem~\ref{thm:lattice}---associative property}
  \\&= x \join \big[ (y\meet x) \join (y \meet z) \big]
    && \text{by Theorem~\ref{thm:lattice}---commutative property}
  \\&= x \join \big[ y\meet(x \join z) \big]
    && \text{by left hypothesis}
  \\&= \mcom{\big[ x \meet (x \join z)\big]}{expand $x$ wrt $z$}  
       \join \big[ y\meet(x \join z) \big]
    && \text{by Theorem~\ref{thm:lattice}---absorptive property}
  \\&= \big[ (x \join z) \meet x\big] \join \big[ (x \join z) \meet y\big]
    && \text{by Theorem~\ref{thm:lattice}---commutative property}
  \\&= (x \join z) \meet (x \join y)
    && \text{by left hypothesis}
  \\&= (x \join y) \meet (x \join z)
    && \text{by Theorem~\ref{thm:lattice}---commutative property}
  \\
\intertext{2. Proof that (left hyothesis) $\impliedby$ (right hypothesis):
  (simply exchange $\join$ and $\meet$ in 1.)}
  x \meet (y \join z)
    &= \mcom{\big[ x \meet (x\join y)\big]}{expand $x$ wrt $y$} \meet (y \join z)
    && \text{by Theorem~\ref{thm:lattice}---absorptive property}
  \\&= x \meet \big[ (x\join y) \meet (y \join z) \big]
    && \text{by Theorem~\ref{thm:lattice}---associative property}
  \\&= x \meet \big[ (y\join x) \meet (y \join z) \big]
    && \text{by Theorem~\ref{thm:lattice}---commutative property}
  \\&= x \meet \big[ y\join(x \meet z) \big]
    && \text{by left hypothesis}
  \\&= \mcom{\big[ x \join (x \meet z)\big]}{expand $x$ wrt $z$}  
       \meet \big[ y\join(x \meet z) \big]
    && \text{by Theorem~\ref{thm:lattice}---absorptive property}
  \\&= \big[ (x \meet z) \join x\big] \meet \big[ (x \meet z) \join y\big]
    && \text{by Theorem~\ref{thm:lattice}---commutative property}
  \\&= (x \meet z) \join (x \meet y)
    && \text{by left hypothesis}
  \\&= (x \meet y) \join (x \meet z)
    && \text{by Theorem~\ref{thm:lattice}---commutative property}
\end{align*}
\end{proof}

%---------------------------------------
\begin{definition}
\label{def:lat_dis}
\citep{burris}{12}
\index{distributive}
%---------------------------------------
\defbox{\begin{array}{l@{\qquad}ll @{\qquad}C @{\qquad}D}
  \mc{5}{l}{\text{A {\bf distributive} lattice is any lattice $(X,\le,\join,\meet)$}} \\
  \mc{5}{l}{\text{that satisfies either (and hence both) of the following:}}
    \\&
    1. & x \meet (y \join z) = (x\meet y) \join (x\meet z)
       & \forall x,y,z\in X
       & ($\meet$ distributes over $\join$)
    \\&
    2. & x \join (y \meet z) = (x\join y) \meet (x\join z)
       & \forall x,y,z\in X
       & ($\join$ distributes over $\meet$)
\end{array}}
\end{definition}


%---------------------------------------
\begin{definition}
\label{def:lat_mod}
\citep{burris}{13}
\index{modular}
%---------------------------------------
\defbox{\begin{array}{l @{\qquad}C}
  \mc{2}{l}{\text{A {\bf modular} lattice is any lattice $(X,\le,\join,\meet)$ that satisfies}}
  \\ \qquad
  x\le y
  \qquad\implies\qquad
  x \join (y\meet z) = y\meet(x\join z)
  & \forall x,y,z \in X
\end{array}}
\end{definition}


%---------------------------------------
\begin{theorem}
\label{def:lat_dis=>mod}
\citep{burris}{13}
\index{distributive}
\index{modular}
%---------------------------------------
Let $(X,\le,\join,\meet)$ be a lattice.
\formbox{
  (X,\le,\join,\meet) \text{ is distributive }
  \qquad\implies\qquad
  (X,\le,\join,\meet) \text{ is modular. }
  }
\end{theorem}
\begin{proof}
\begin{align*}
  x \join (y\meet z)
    &= (x\join y) \meet (x\join z)
    && \text{by distributive hypothesis}
  \\&= y \meet (x\join z)
    && \text{by $x\le y$ hypothesis}
\end{align*}
\end{proof}
