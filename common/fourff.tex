\chapter{Fourier Transform on Finite Fields}
%=======================================

%---------------------------------------
\begin{definition}[Fourier transform over a finite field]
\citetbl{
  \citerp{pollard1971}{365} \\
  \citerp{caire1993}{1158}
  }
\label{def:fff}
%---------------------------------------
Let $GF(p^n)$ be the finite field over $p^n$ where $p$ is prime and $p\in\Zp$.
\defboxp{
  $\ds \opF \seqn{x_n} \eqd \sum_{n=0}^{d-1} x_n r^{nm}$
  }

\end{definition}

%---------------------------------------
\begin{definition}[Inverse Fourier transform]
\citetbl{
  \citerp{pollard1971}{365}
  }
\label{def:fff_inv}
%---------------------------------------
Let \seq{x_n\in F}{n\in\Z} be a sequence over a field $F$.
\defboxp{
  $\ds \opFi\seqn{x_n} \eqd -d'\sum_{n=0}^{d-1} x_n r^{-nm}$
  where $d'$ is the element such that $d'd=1$
  }
\end{definition}

%---------------------------------------
\begin{theorem}
%---------------------------------------
\defboxp{
  $\ds \opF\opFi = \opFi\opF = \opI $
  }
\end{theorem}
\begin{proof}
\begin{align*}
  \opFi\opF\seqn{x_n}
    &= \opFi \brp{\sum_{n=0}^{d-1} x_n r^{nm}}
    && \text{by \prefp{def:fff}}
  \\&= -d'\sum_{m=0}^{d-1} \brp{\sum_{n=0}^{d-1} x_n r^{nm}}   r^{-mk}
    && \text{by \prefp{def:fff_inv}}
  \\&= -d'\sum_{m=0}^{d-1} \sum_{n=0}^{d-1} x_n r^{nm}   r^{-mk}
  \\&= \sum_{n=0}^{d-1} x_n (-d') \sum_{m=0}^{d-1}  r^{m(n-k)} 
\end{align*}
\end{proof}

%---------------------------------------
\begin{example}[Discrete fourier transform]
\citetbl{
  \citerp{caire1993}{1158}
  }
\label{ex:fff_dft}
%---------------------------------------
\exboxp{
  $\ds \opF  \seqn{x_n} \eqd \sum_{n=0}^{d-1} x_n \mcom{\brp{e^{i\frac{2\pi}{N}}}}{$r$}^{nm}$
  \qquad\qquad
  $\ds \opFi \seqn{x_n} \eqd \mcom{\frac{1}{N}}{$d'$} \sum_{n=0}^{d-1} x_n \mcom{\brp{e^{i\frac{2\pi}{N}}}}{$r$}^{(-nm)}$
  }
\end{example}
